  <action>
Create a new file `src/app/_components/landing/HowItWorksAnimationShell.tsx`.

This Client Component:
1. Attaches a rootRef to a wrapper div
2. On mount, registers a createScope scoped to rootRef
3. Inside the scope, sets up two behaviors:
   a. Scroll reveal: onScroll targeting rootRef.current, onEnterForward callback that staggers `.how-it-works-card` elements in from y:50 with opacity 0→1, 700ms duration, 90ms stagger delay, outQuint ease. Play-once (onEnterForward guarantees this).
   b. Hover lift: For each `.how-it-works-card`, add mouseenter/mouseleave listeners that animate translateY to -5 and boxShadow to the hover value on enter, and reverse on leave.
4. Cleanup via scope.current?.revert() returns all animations and event listeners.

The wrapper div should use `className="contents"` — this makes it layout-neutral so it does not affect the grid rendering of HowItWorksSection (the section itself is the layout owner).

Full implementation:

```typescript
'use client';

import { useEffect, useRef } from 'react';
import { createScope, animate, onScroll, stagger } from 'animejs';

export function HowItWorksAnimationShell({ children }: { children: React.ReactNode }) {
  const rootRef = useRef<HTMLDivElement>(null);
  const scope = useRef<ReturnType<typeof createScope> | null>(null);

  useEffect(() => {
    if (!rootRef.current) return;

    scope.current = createScope({ root: rootRef }).add(() => {
      // Scroll-triggered stagger reveal (ANIM-01)
      onScroll({
        target: rootRef.current!,
        enter: 'top bottom',
        onEnterForward: () => {
          animate('.how-it-works-card', {
            opacity: [0, 1],
            translateY: [50, 0],
            duration: 700,
            delay: stagger(90),
            ease: 'outQuint',
          });
        },
      });

      // Hover lift on each card (ANIM-04)
      const cards = rootRef.current!.querySelectorAll<HTMLElement>('.how-it-works-card');
      cards.forEach((card) => {
        card.addEventListener('mouseenter', () => {
          animate(card, {
            translateY: -5,
            boxShadow: '0 8px 24px rgba(1,30,65,0.10), 0 4px 8px rgba(1,30,65,0.06)',
            duration: 250,
            ease: 'outQuad',
          });
        });
        card.addEventListener('mouseleave', () => {
          animate(card, {
            translateY: 0,
            boxShadow: '0 4px 8px rgba(1,30,65,0.06), 0 1px 3px rgba(1,30,65,0.04)',
            duration: 250,
            ease: 'outQuad',
          });
        });
      });
    });

    return () => scope.current?.revert();
  }, []);

  return (
    <div ref={rootRef} className="contents">
      {children}
    </div>
  );
}
```

Note: The boxShadow start and end values each have exactly 2 shadow layers (comma-separated). This is required for Anime.js to interpolate between them — different layer counts cause a snap instead of a tween.

**FeatureStatsAnimationShell.tsx:**

Create `src/app/_components/landing/FeatureStatsAnimationShell.tsx`.

This shell provides:
- Scroll reveal: onScroll with onEnterForward that staggers the `.stat-number` wrappers in (opacity 0→1, y:40→0, 600ms, 100ms stagger, outQuint).
- Count-up: Inside the same onEnterForward callback (after stagger), animate each `.stat-value` element's innerHTML from 0 to its `data-value` with outExpo ease and 1800ms duration and `round: 1`.

Important: Both animations fire together in onEnterForward. The stagger entrance and the count-up start simultaneously — the count-up duration (1800ms) is longer than the stagger entrance (600ms + stagger offsets) so the numbers finish counting after the cards have appeared.

```typescript
'use client';

import { useEffect, useRef } from 'react';
import { createScope, animate, onScroll, stagger } from 'animejs';

export function FeatureStatsAnimationShell({ children }: { children: React.ReactNode }) {
  const rootRef = useRef<HTMLDivElement>(null);
  const scope = useRef<ReturnType<typeof createScope> | null>(null);

  useEffect(() => {
    if (!rootRef.current) return;

    scope.current = createScope({ root: rootRef }).add(() => {
      onScroll({
        target: rootRef.current!,
        enter: 'top bottom',
        onEnterForward: () => {
          // Stagger entrance of stat cards (ANIM-01)
          animate('.stat-number', {
            opacity: [0, 1],
            translateY: [40, 0],
            duration: 600,
            delay: stagger(100),
            ease: 'outQuint',
          });

          // Count-up on numeric value spans (ANIM-03)
          const statValues = rootRef.current!.querySelectorAll<HTMLElement>('.stat-value');
          statValues.forEach((el) => {
            const endValue = parseInt(el.dataset.value ?? '0', 10);
            animate(el, {
              innerHTML: [0, endValue],
              round: 1,
              duration: 1800,
              ease: 'outExpo',
            });
          });
        },
      });
    });

    return () => scope.current?.revert();
  }, []);

  return (
    <div ref={rootRef} className="contents">
      {children}
    </div>
  );
}
```

**HeroAnimationShell.tsx:**

Create `src/app/_components/landing/HeroAnimationShell.tsx`.

This shell provides the CTA button breathing amber glow (ANIM-04). No scroll reveal — Hero is already in viewport on load.

The breathing glow is a loop:true + alternate:true animation that pulses box-shadow between a dim and bright amber value while the user hovers. On mouseleave, the loop animation is paused and a short fade-out animation resets the shadow.

```typescript
'use client';

import { useEffect, useRef } from 'react';
import { createScope, animate } from 'animejs';
import type { Animatable } from 'animejs';

export function HeroAnimationShell({ children }: { children: React.ReactNode }) {
  const rootRef = useRef<HTMLDivElement>(null);
  const scope = useRef<ReturnType<typeof createScope> | null>(null);

  useEffect(() => {
    if (!rootRef.current) return;

    scope.current = createScope({ root: rootRef }).add(() => {
      const ctaButton = rootRef.current!.querySelector<HTMLElement>('.cta-button');
      if (!ctaButton) return;

      let glowAnimation: Animatable | null = null;

      ctaButton.addEventListener('mouseenter', () => {
        glowAnimation = animate(ctaButton, {
          boxShadow: [
            '0 4px 16px rgba(245,168,0,0.20)',
            '0 6px 24px rgba(245,168,0,0.50)',
          ],
          duration: 900,
          loop: true,
          alternate: true,
          ease: 'inOutSine',
        });
      });

      ctaButton.addEventListener('mouseleave', () => {
        if (glowAnimation) {
          glowAnimation.pause();
          glowAnimation = null;
        }
        animate(ctaButton, {
          boxShadow: '0 4px 16px rgba(245,168,0,0.00)',
          duration: 300,
          ease: 'outQuad',
        });
      });
    });

    return () => scope.current?.revert();
  }, []);

  return (
    <div ref={rootRef} className="contents">
      {children}
    </div>
  );
}
```

Note on `Animatable` type: If `Animatable` is not exported from animejs, use `ReturnType<typeof animate>` instead. Check the animejs package types at implementation time and use whichever resolves correctly. Do not use `any`.
