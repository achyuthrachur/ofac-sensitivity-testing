    - 'abc' (NaN) → 0
    - '10' → 10
    - '500' → 500
    - '501' (above max) → 500 (clamped to MAX_ENTITY_COUNT)
    - '-5' (negative) → 0

    toggleRegion(current: Region[], target: Region): Region[]
    - ['arabic', 'cjk'] + 'arabic' → ['cjk'] (removes present)
    - ['arabic'] + 'cjk' → ['arabic', 'cjk'] (adds missing)
    - [] + 'arabic' → ['arabic']
    - Does not mutate the input array

    toggleRule(current: string[], target: string): string[]
    - ['space-removal'] + 'space-removal' → [] (removes present)
    - ['space-removal'] + 'char-substitution' → ['space-removal', 'char-substitution'] (adds)
    - Does not mutate the input array

    deriveSelectAllState(ruleIds: string[]): boolean | 'indeterminate'
