    apiRef.current = Comlink.wrap<ScreeningWorkerApi>(workerRef.current);

  return () => {
    try {
      if (apiRef.current) apiRef.current[Comlink.releaseProxy]();
      workerRef.current?.terminate();
    } catch { /* cleanup errors are non-critical */ }
  };
}, []);

        const effectiveTier = r.nameLengthPenaltyApplied ? escalateTier(riskTier) : riskTier;
        return { ...r, riskTier, effectiveTier };
      });

    IMPORTANT: `escalateTier` is needed for the penalty escalation. Define it locally in this file or import from scorer.ts if it is a pure export. Check if scorer.ts exports it; if not, define it inline:
    ```typescript
    const TIER_ORDER: RiskTier[] = ['CLEAR', 'LOW', 'MEDIUM', 'HIGH', 'EXACT'];
    function escalateTier(tier: RiskTier): RiskTier {
      const i = TIER_ORDER.indexOf(tier);
      return i < TIER_ORDER.length - 1 ? TIER_ORDER[i + 1] : tier;
    }
    ```

    useMemo — tierCounts (for the inline summary bar):
    ```typescript
    const tierCounts = useMemo(() => {
      const counts: Record<RiskTier, number> = { EXACT: 0, HIGH: 0, MEDIUM: 0, LOW: 0, CLEAR: 0 };
      for (const r of displayResults) counts[r.effectiveTier]++;
      return counts;
    }, [displayResults]);
    ```
